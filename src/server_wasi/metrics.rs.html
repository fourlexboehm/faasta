20#[derive(Debug)]
21pub struct FunctionMetric {
22    pub function_name: String,
23    pub total_time: AtomicU64,
24    pub call_count: AtomicU64,
25    pub last_called: AtomicU64,
26}

28// Manual implementation of Clone for FunctionMetric
29impl Clone for FunctionMetric {
30    fn clone(&self) -> Self {
31        Self {
32            function_name: self.function_name.clone(),
33            total_time: AtomicU64::new(self.total_time.load(Ordering::Relaxed)),
34            call_count: AtomicU64::new(self.call_count.load(Ordering::Relaxed)),
35            last_called: AtomicU64::new(self.last_called.load(Ordering::Relaxed)),
36        }
37    }
38}
39
40impl FunctionMetric {
41    pub fn new(function_name: String) -> Self {
42        // Initialize the last_called timestamp to current time
43        let now = SystemTime::now()
44            .duration_since(UNIX_EPOCH)
45            .unwrap_or(Duration::from_secs(0))
46            .as_millis() as u64;
47
48        // Try to load from sled if it exists
49        let metric = if let Ok(Some(data)) = METRICS_DB.get(function_name.as_bytes()) {
50            if let Ok(((total_time, call_count, last_called), _)) =
51                bincode::decode_from_slice::<(u64, u64, u64), bincode::config::Configuration>(
52                    &data,
53                    bincode::config::standard(),
54                )
55            {
56                Self {
57                    function_name,
58                    total_time: AtomicU64::new(total_time),
59                    call_count: AtomicU64::new(call_count),
60                    last_called: AtomicU64::new(last_called),
61                }
62            } else {
63                Self::default(function_name, now)
64            }
65        } else {
66            Self::default(function_name, now)
67        };
68
69        debug!(
70            "Created or loaded metric for function: {}",
71            metric.function_name
72        );
73        metric
74    }
75
76    fn default(function_name: String, now: u64) -> Self {
77        Self {
78            function_name,
79            total_time: AtomicU64::new(0),
80            call_count: AtomicU64::new(0),
81            last_called: AtomicU64::new(now),
82        }
83    }
84
85    pub fn record_call(&self, duration_ms: u64) {
86        // Update in-memory metrics
87        let prev_total = self.total_time.fetch_add(duration_ms, Ordering::Relaxed);
88        let prev_count = self.call_count.fetch_add(1, Ordering::Relaxed);
89
90        // Update last called timestamp (milliseconds since epoch)
91        let now = SystemTime::now()
92            .duration_since(UNIX_EPOCH)
93            .unwrap_or(Duration::from_secs(0))
94            .as_millis() as u64;
95        self.last_called.store(now, Ordering::Relaxed);
96
97        // Log the metrics update with more detailed information
98        debug!(
99            "Recorded metrics for function '{}': duration={}ms, prev_total={}ms, new_total={}ms, prev_calls={}, new_calls={}",
100            self.function_name,
101            duration_ms,
102            prev_total,
103            prev_total + duration_ms,
104            prev_count,
105            prev_count + 1
106        );
107
108        // No immediate persistence; metrics will be flushed periodically
109    }
110
111    // Method to flush this individual function's metrics to the database
112    pub fn flush_to_db(&self) {
113        // Load existing DB values
114        let existing = METRICS_DB
115            .get(self.function_name.as_bytes())
116            .unwrap_or(None);
117        let (db_total, db_calls, db_last) = if let Some(db_bytes) = existing {
118            if let Ok(((t, c, l), _)) = bincode::decode_from_slice::<
119                (u64, u64, u64),
120                bincode::config::Configuration,
121            >(&db_bytes, bincode::config::standard())
122            {
123                info!(
124                    "Found existing DB metrics for '{}': total={}ms, calls={}, last={}",
125                    self.function_name, t, c, l
126                );
127                (t, c, l)
128            } else {
129                info!(
130                    "Failed to decode existing DB metrics for '{}', using zeros",
131                    self.function_name
132                );
133                (0, 0, 0)
134            }
135        } else {
136            info!(
137                "No existing DB metrics for '{}', using zeros",
138                self.function_name
139            );
140            (0, 0, 0)
141        };
142
143        // Add current in-memory values
144        let mem_total = self.total_time.load(Ordering::Relaxed);
145        let mem_calls = self.call_count.load(Ordering::Relaxed);
146        let mem_last = self.last_called.load(Ordering::Relaxed);
147
148        info!(
149            "In-memory metrics for '{}': total={}ms, calls={}, last={}",
150            self.function_name, mem_total, mem_calls, mem_last
151        );
152
153        // Calculate combined values
154        let combined_total = db_total + mem_total;
155        let combined_calls = db_calls + mem_calls;
156        let combined_last = std::cmp::max(db_last, mem_last);
157
158        info!(
159            "Combined metrics for '{}': total={}ms, calls={}, last={}",
160            self.function_name, combined_total, combined_calls, combined_last
161        );
162
163        // Combine and persist
164        if let Ok(data) = bincode::encode_to_vec(
165            (combined_total, combined_calls, combined_last),
166            bincode::config::standard(),
167        ) {
168            match METRICS_DB.insert(self.function_name.as_bytes(), data) {
169                Ok(_) => info!(
170                    "Successfully persisted metrics for '{}'",
171                    self.function_name
172                ),
173                Err(e) => error!(
174                    "Failed to persist metrics for '{}': {}",
175                    self.function_name, e
176                ),
177            }
178        } else {
179            error!("Failed to encode metrics for '{}'", self.function_name);
180        }
181    }
182}

184// Function to check if a function's WASM file exists
185fn function_wasm_exists(function_name: &str) -> bool {
186    // Get the functions directory from environment or use default
187    let functions_dir =
188        std::env::var("FUNCTIONS_PATH").unwrap_or_else(|_| "./functions".to_string());
189
190    let wasm_filename = format!("{}.wasm", function_name);
191    let wasm_path = Path::new(&functions_dir).join(&wasm_filename);
192
193    wasm_path.exists()
194}

196pub fn get_metrics() -> Metrics {
197    info!("Retrieving metrics from database...");
198    let mut function_metrics = Vec::new();
199    let mut total_time = 0;
200    let mut total_calls = 0;
201
202    // Log the number of entries in the metrics database
203    let db_entries_count = METRICS_DB.iter().count();
204    info!("Found {} entries in metrics database", db_entries_count);
205
206    for (key, value) in METRICS_DB.iter().flatten() {
207        if key.starts_with(b"user:") {
208            continue;
209        }
210
211        let function_name = match str::from_utf8(&key) {
212            Ok(name) => name.to_string(),
213            Err(_) => {
214                debug!("Skipping invalid UTF-8 key in metrics database");
215                continue; // Skip invalid UTF-8 keys
216            }
217        };
218
219        // Decode the DB metrics data
220        if let Ok(((db_total_time, db_call_count, db_last_called), _)) =
221            bincode::decode_from_slice::<(u64, u64, u64), bincode::config::Configuration>(
222                &value,
223                bincode::config::standard(),
224            )
225        {
226            info!(
227                "DB metrics for '{}': total={}ms, calls={}, last={}",
228                function_name, db_total_time, db_call_count, db_last_called
229            );
230
231            // Load in-memory metrics
232            let (mem_total_time, mem_call_count, mem_last_called) = FUNCTION_METRICS
233                .get(&function_name)
234                .map(|m| {
235                    let total = m.total_time.load(Ordering::Relaxed);
236                    let calls = m.call_count.load(Ordering::Relaxed);
237                    let last = m.last_called.load(Ordering::Relaxed);
238
239                    info!(
240                        "In-memory metrics for '{}': total={}ms, calls={}, last={}",
241                        function_name, total, calls, last
242                    );
243
244                    (total, calls, last)
245                })
246                .unwrap_or_else(|| {
247                    info!("No in-memory metrics for '{}', using zeros", function_name);
248                    (0, 0, 0)
249                });
250
251            // Combine DB and in-memory metrics
252            let combined_total_time = db_total_time.saturating_add(mem_total_time);
253            let combined_call_count = db_call_count.saturating_add(mem_call_count);
254            let combined_last_called = std::cmp::max(db_last_called, mem_last_called);
255
256            info!(
257                "Combined metrics for '{}': total={}ms, calls={}, last={}",
258                function_name, combined_total_time, combined_call_count, combined_last_called
259            );
260
261            // Convert timestamp to ISO string
262            let last_called_time = UNIX_EPOCH + Duration::from_millis(combined_last_called);
263            let last_called_str =
264                chrono::DateTime::<chrono::Utc>::from(last_called_time).to_rfc3339();
265
266            function_metrics.push(FunctionMetricsResponse {
267                function_name: function_name.clone(),
268                total_time_millis: combined_total_time,
269                call_count: combined_call_count,
270                last_called: last_called_str,
271            });
272
273            total_time += combined_total_time;
274            total_calls += combined_call_count;
275        } else {
276            error!(
277                "Failed to decode metrics data for function '{}'",
278                function_name
279            );
280        }
281    }
282
283    info!(
284        "Returning metrics: {} functions, {} total calls, {} total ms",
285        function_metrics.len(),
286        total_calls,
287        total_time
288    );
289
290    Metrics {
291        total_time,
292        total_calls,
293        function_metrics,
294    }
295}

297// Helper function to get or create a function metric
298pub fn get_or_create_metric(function_name: &str) -> Option<FunctionMetric> {
299    // Use entry API to reduce lock contention
300    let entry = FUNCTION_METRICS.entry(function_name.to_string());
301
302    match entry {
303        dashmap::mapref::entry::Entry::Occupied(occupied) => {
304            // Return a clone of the existing metric
305            Some(FunctionMetric::new(occupied.key().clone()))
306        }
307        dashmap::mapref::entry::Entry::Vacant(vacant) => {
308            // First check if the function's WASM file exists
309            if !function_wasm_exists(function_name) {
310                return None;
311            }
312
313            debug!("Creating new metric for function: {}", function_name);
314
315            // Create the new metric
316            let metric = FunctionMetric::new(function_name.to_string());
317
318            // Insert it into the map
319            vacant.insert(metric.clone());
320
321            // New function added - ensure it's recorded in Sled DB even if no calls happen
322            if !METRICS_DB
323                .contains_key(function_name.as_bytes())
324                .unwrap_or(false)
325            {
326                // Get current time in milliseconds for initialization
327                let now = SystemTime::now()
328                    .duration_since(UNIX_EPOCH)
329                    .unwrap_or(Duration::from_secs(0))
330                    .as_millis() as u64;
331
332                // Initialize with zeros
333                let initial_data =
334                    match bincode::encode_to_vec((0u64, 0u64, now), bincode::config::standard()) {
335                        Ok(data) => data,
336                        Err(e) => {
337                            error!(
338                                "Failed to encode initial metrics for new function {}: {}",
339                                function_name, e
340                            );
341                            return None;
342                        }
343                    };
344                let _ = METRICS_DB.insert(function_name.as_bytes(), initial_data);
345                debug!("Added new function '{}' to metrics database", function_name);
346            }
347
348            Some(metric)
349        }
350    }
351}

353// Timer utility to measure function execution time
354pub struct Timer {
355    start: SystemTime,
356    function_name: String,
357}
358
359impl Timer {
360    #[tracing::instrument(level = "debug")]
361    pub fn new(function_name: String) -> Self {
362        Self {
363            start: SystemTime::now(),
364            function_name,
365        }
366    }
367}
368
369impl Drop for Timer {
370    fn drop(&mut self) {
371        let duration = SystemTime::now()
372            .duration_since(self.start)
373            .unwrap_or(Duration::from_secs(0));
374
375        if let Some(metric) = get_or_create_metric(&self.function_name) {
376            // Round up any duration to at least 1 millisecond
377            let duration_ms = duration.as_millis() as u64;
378            // Ensure the minimum duration is 1ms, even if the actual duration was 0ms
379            let rounded_duration = std::cmp::max(duration_ms, 1);
380
381            metric.record_call(rounded_duration);
382        }
383    }
384}

386/// Flush in-memory metrics to persistent DB and reset counters.
387pub fn flush_metrics_to_db() {
388    info!("Flushing metrics to database...");
389    let mut flushed_count = 0;
390
391    for entry in FUNCTION_METRICS.iter() {
392        let metric = entry.value(); // We only need the metric, not the key
393        let function_name = &metric.function_name;
394        let call_count = metric.call_count.load(Ordering::Relaxed);
395        let total_time = metric.total_time.load(Ordering::Relaxed);
396
397        // Skip if no calls were made since last flush
398        if call_count == 0 {
399            debug!(
400                "Skipping flush for function '{}' - no calls since last flush",
401                function_name
402            );
403            continue; // Skip if no calls were made
404        }
405
406        info!(
407            "Flushing metrics for function '{}': calls={}, total_time={}ms",
408            function_name, call_count, total_time
409        );
410
411        // First flush this function's current metrics to the database
412        // using our helper method
413        metric.flush_to_db();
414
415        // Then reset the in-memory counters
416        metric.total_time.store(0, Ordering::Relaxed);
417        metric.call_count.store(0, Ordering::Relaxed);
418
419        // Don't reset last_called timestamp
420        // This preserves when the function was last used even after resetting counters
421
422        flushed_count += 1;
423    }
424
425    if flushed_count > 0 {
426        // Ensure DB writes are durable
427        if let Err(e) = METRICS_DB.flush() {
428            error!("Failed to flush metrics DB: {}", e);
429        } else {
430            info!(
431                "Successfully flushed metrics for {} functions",
432                flushed_count
433            );
434        }
435    } else {
436        // Log when no metrics were flushed (for monitoring)
437        debug!("No metrics to flush - no functions were called since last flush");
438    }
439}

441/// Spawn a Tokio task to periodically flush metrics to DB every `interval_secs` seconds.
442pub fn spawn_periodic_flush(interval_secs: u64) {
443    tokio::spawn(async move {
444        let mut ticker = interval(TokioDuration::from_secs(interval_secs));
445        loop {
446            ticker.tick().await;
447            flush_metrics_to_db();
448        }
449    });
