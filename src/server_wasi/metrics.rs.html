13// Sled database for persistent storage
14pub static METRICS_DB: Lazy<sled::Db> = Lazy::new(|| {
15    let db_path = std::env::var("METRICS_DB_PATH").unwrap_or_else(|_| "./data/metrics".to_string());
16    sled::open(db_path).expect("Failed to open metrics database")

19#[derive(Debug)]
20pub struct FunctionMetric {
21    pub function_name: String,
22    pub total_time: AtomicU64,
23    pub call_count: AtomicU64,
24    pub last_called: AtomicU64,
25}
26
27impl FunctionMetric {
28    pub fn new(function_name: String) -> Self {
29        // Try to load from sled if it exists
30        let metric = if let Ok(Some(data)) = METRICS_DB.get(function_name.as_bytes()) {
31            if let Ok(((total_time, call_count, last_called), _)) =
32                bincode::decode_from_slice::<(u64, u64, u64), bincode::config::Configuration>(
33                    &data,
34                    bincode::config::standard(),
35                )
36            {
37                Self {
38                    function_name,
39                    total_time: AtomicU64::new(total_time),
40                    call_count: AtomicU64::new(call_count),
41                    last_called: AtomicU64::new(last_called),
42                }
43            } else {
44                Self::default(function_name)
45            }
46        } else {
47            Self::default(function_name)
48        };
49
50        metric
51    }
52
53    fn default(function_name: String) -> Self {
54        Self {
55            function_name,
56            total_time: AtomicU64::new(0),
57            call_count: AtomicU64::new(0),
58            last_called: AtomicU64::new(0),
59        }
60    }
61
62    pub fn record_call(&self, duration_ms: u64) {
63        // Update in-memory metrics
64        self.total_time.fetch_add(duration_ms, Ordering::Relaxed);
65        self.call_count.fetch_add(1, Ordering::Relaxed);
66
67        // Update last called timestamp (milliseconds since epoch)
68        let now = SystemTime::now()
69            .duration_since(UNIX_EPOCH)
70            .unwrap_or(Duration::from_secs(0))
71            .as_millis() as u64;
72        self.last_called.store(now, Ordering::Relaxed);
73
74        // Log the metrics update
75        debug!(
76            "Recorded metrics for function '{}': duration={}ms, total_calls={}, total_time={}ms",
77            self.function_name,
78            duration_ms,
79            self.call_count.load(Ordering::Relaxed),
80            self.total_time.load(Ordering::Relaxed)
81        );
82
83        // No immediate persistence; metrics will be flushed periodically
84    }
85
86    // Method to flush this individual function's metrics to the database
87    pub fn flush_to_db(&self) {
88        // Load existing DB values
89        let existing = METRICS_DB
90            .get(self.function_name.as_bytes())
91            .unwrap_or(None);
92        let (db_total, db_calls, db_last) = if let Some(db_bytes) = existing {
93            if let Ok(((t, c, l), _)) = bincode::decode_from_slice::<
94                (u64, u64, u64),
95                bincode::config::Configuration,
96            >(&db_bytes, bincode::config::standard())
97            {
98                (t, c, l)
99            } else {
100                (0, 0, 0)
101            }
102        } else {
103            (0, 0, 0)
104        };
105
106        // Add current in-memory values
107        let mem_total = self.total_time.load(Ordering::Relaxed);
108        let mem_calls = self.call_count.load(Ordering::Relaxed);
109        let mem_last = self.last_called.load(Ordering::Relaxed);
110
111        // Combine and persist
112        if let Ok(data) = bincode::encode_to_vec(
113            (
114                db_total + mem_total,
115                db_calls + mem_calls,
116                std::cmp::max(db_last, mem_last),
117            ),
118            bincode::config::standard(),
119        ) {
120            let _ = METRICS_DB.insert(self.function_name.as_bytes(), data);
121        }
122    }
123}
124pub fn get_metrics() -> Metrics {
125    let mut function_metrics = Vec::new();
126    let mut total_time = 0;
127    let mut total_calls = 0;
128
129    // Iterate through all entries in the sled database
130    for (key, value) in METRICS_DB.iter().flatten() {
131        // Skip user project count entries (they start with "user:")
132        if key.starts_with(b"user:") {
133            continue;
134        }
135
136        let function_name = match str::from_utf8(&key) {
137            Ok(name) => name.to_string(),
138            Err(_) => continue, // Skip invalid UTF-8 keys
139        };
140
141        // Decode the DB metrics data
142        if let Ok(((db_total_time, db_call_count, db_last_called), _)) =
143            bincode::decode_from_slice::<(u64, u64, u64), bincode::config::Configuration>(
144                &value,
145                bincode::config::standard(),
146            )
147        {
148            // Load in-memory metrics
149            let (mem_total_time, mem_call_count, mem_last_called) = FUNCTION_METRICS
150                .get(&function_name)
151                .map(|m| {
152                    (
153                        m.total_time.load(Ordering::Relaxed),
154                        m.call_count.load(Ordering::Relaxed),
155                        m.last_called.load(Ordering::Relaxed),
156                    )
157                })
158                .unwrap_or((0, 0, 0));
159
160            // Combine DB and in-memory metrics
161            let combined_total_time = db_total_time.saturating_add(mem_total_time);
162            let combined_call_count = db_call_count.saturating_add(mem_call_count);
163            let combined_last_called = std::cmp::max(db_last_called, mem_last_called);
164
165            // Convert timestamp to ISO string
166            let last_called_time = UNIX_EPOCH + Duration::from_millis(combined_last_called);
167            let last_called_str =
168                chrono::DateTime::<chrono::Utc>::from(last_called_time).to_rfc3339();
169
170            function_metrics.push(FunctionMetricsResponse {
171                function_name: function_name.clone(),
172                total_time_millis: combined_total_time,
173                call_count: combined_call_count,
174                last_called: last_called_str,
175            });
176
177            total_time += combined_total_time;
178            total_calls += combined_call_count;
179        }
180    }
181
182    info!(
183        "Returning metrics: {} functions, {} total calls, {} total ms",
184        function_metrics.len(),
185        total_calls,
186        total_time
187    );
188
189    Metrics {
190        total_time,
191        total_calls,
192        function_metrics,
193    }
194}

196// Helper function to get or create a function metric
197pub fn get_or_create_metric(function_name: &str) -> FunctionMetric {
198    // Check if function exists in memory cache
199    let is_new_function = !FUNCTION_METRICS.contains_key(function_name);
200
201    if is_new_function {
202        // Create the new metric
203        let metric = FunctionMetric::new(function_name.to_string());
204        FUNCTION_METRICS.insert(function_name.to_string(), metric);
205
206        // New function added - ensure it's recorded in Sled DB even if no calls happen
207        // This is important for tracking deployed functions even before any calls
208        if !METRICS_DB
209            .contains_key(function_name.as_bytes())
210            .unwrap_or(false)
211        {
212            // Initialize with zeros
213            let initial_data =
214                match bincode::encode_to_vec((0u64, 0u64, 0u64), bincode::config::standard()) {
215                    Ok(data) => data,
216                    Err(e) => {
217                        error!(
218                            "Failed to encode initial metrics for new function {}: {}",
219                            function_name, e
220                        );
221                        return FunctionMetric::new(function_name.to_string());
222                    }
223                };
224            let _ = METRICS_DB.insert(function_name.as_bytes(), initial_data);
225            debug!("Added new function '{}' to metrics database", function_name);
226        }
227    }
228    // Get a copy of the metric
229    let entry = FUNCTION_METRICS.get(function_name).unwrap();
230    FunctionMetric::new(entry.function_name.clone())
231}

233// Timer utility to measure function execution time
234pub struct Timer {
235    start: SystemTime,
236    function_name: String,
237}
238
239impl Timer {
240    pub fn new(function_name: String) -> Self {
241        Self {
242            start: SystemTime::now(),
243            function_name,
244        }
245    }
246}
247
248impl Drop for Timer {
249    fn drop(&mut self) {
250        let duration = SystemTime::now()
251            .duration_since(self.start)
252            .unwrap_or(Duration::from_secs(0));
253
254        let metric = get_or_create_metric(&self.function_name);
255        metric.record_call(duration.as_millis() as u64);
256    }
257}

259/// Flush in-memory metrics to persistent DB and reset counters.
260pub fn flush_metrics_to_db() {
261    info!("Flushing metrics to database...");
262    let mut flushed_count = 0;
263
264    for entry in FUNCTION_METRICS.iter() {
265        let metric = entry.value(); // We only need the metric, not the key
266
267        // Skip if no calls were made since last flush
268        if metric.call_count.load(Ordering::Relaxed) == 0 {
269            continue; // Skip if no calls were made
270        }
271
272        // First flush this function's current metrics to the database
273        // using our helper method
274        metric.flush_to_db();
275
276        // Then reset the in-memory counters
277        metric.total_time.store(0, Ordering::Relaxed);
278        metric.call_count.store(0, Ordering::Relaxed);
279
280        // Don't reset last_called timestamp
281        // This preserves when the function was last used even after resetting counters
282
283        flushed_count += 1;
284    }
285
286    if flushed_count > 0 {
287        // Ensure DB writes are durable
288        if let Err(e) = METRICS_DB.flush() {
289            error!("Failed to flush metrics DB: {}", e);
290        } else {
291            info!(
292                "Successfully flushed metrics for {} functions",
293                flushed_count
294            );
295        }
296    } else {
297        // Log when no metrics were flushed (for monitoring)
298        debug!("No metrics to flush - no functions were called since last flush");
299    }
300}

302/// Spawn a Tokio task to periodically flush metrics to DB every `interval_secs` seconds.
303pub fn spawn_periodic_flush(interval_secs: u64) {
304    tokio::spawn(async move {
305        let mut ticker = interval(TokioDuration::from_secs(interval_secs));
306        loop {
307            ticker.tick().await;
308            flush_metrics_to_db();
309        }
310    });
