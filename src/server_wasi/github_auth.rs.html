1use anyhow::Result;
2use bincode::{Decode, Encode};
3use dashmap::DashMap;
4use serde::{Deserialize, Serialize};
5use serde_json::Value;
6
7const USER_DB_TREE: &str = "user_data";
8const MAX_PROJECTS_PER_USER: usize = 10;
9
10/// Struct to hold GitHub auth configuration
11pub struct GitHubAuth {
12    user_projects: DashMap<String, UserData>,
13    db: sled::Db,
14}
15#[derive(Serialize, Deserialize, Clone, Debug, Encode, Decode)]
16pub struct UserData {
17    pub github_username: String,
18    pub projects: Vec<String>,
19}
20
21impl GitHubAuth {
22    pub async fn new(db: sled::Db) -> Result<Self> {
23        // Load existing user data
24        let user_projects = DashMap::new();
25
26        // Create or get the user data tree
27        let user_tree = db.open_tree(USER_DB_TREE)?;
28
29        // Iterate through all items in the tree
30        for item in user_tree.iter().flatten() {
31            if let Ok(username) = std::str::from_utf8(&item.0) {
32                // Try to decode using bincode
33                if let Ok((user_data, _)) =
34                    bincode::decode_from_slice::<UserData, _>(&item.1, bincode::config::standard())
35                {
36                    user_projects.insert(username.to_string(), user_data);
37                } else {
38                    // Fallback to serde_json for backward compatibility
39                    if let Ok(user_data) = serde_json::from_slice::<UserData>(&item.1) {
40                        user_projects.insert(username.to_string(), user_data);
41                    }
42                }
43            }
44        }
45
46        Ok(Self { user_projects, db })
47    }
48
49    /// Validate OAuth token directly with GitHub API
50    pub async fn validate_oauth_token(&self, username: &str, token: &str) -> Result<bool> {
51        // Extract token from "Bearer {token}" format if present
52        let token_value = token.strip_prefix("Bearer ").unwrap_or(token).trim();
53
54        // Create client to verify with GitHub API
55        let client = reqwest::Client::new();
56        let response = client
57            .get("https://api.github.com/user")
58            .header("User-Agent", "faasta-server")
59            .header("Authorization", format!("Bearer {}", token_value))
60            .send()
61            .await?;
62
63        if !response.status().is_success() {
64            return Ok(false);
65        }
66
67        // Verify username matches
68        let github_user: Value = response.json().await?;
69        let api_username = github_user["login"].as_str().unwrap_or("");
70
71        Ok(api_username == username)
72    }
73
74    /// Check if a user can upload more projects (limit is MAX_PROJECTS_PER_USER)
75    pub fn can_upload_project(&self, username: &str, project_name: &str) -> bool {
76        if let Some(user_data) = self.user_projects.get(username) {
77            // Check if they're already at the limit
78            if user_data.projects.len() >= MAX_PROJECTS_PER_USER
79                && !user_data.projects.contains(&project_name.to_string())
80            {
81                return false;
82            }
83        }
84        true
85    }
86
87    /// Add a project to a user's list
88    pub async fn add_project(&self, username: &str, project_name: &str) -> Result<()> {
89        // Get or create user data
90        let mut user_data = if let Some(data) = self.user_projects.get(username) {
91            data.clone()
92        } else {
93            UserData {
94                github_username: username.to_string(),
95                projects: Vec::new(),
96            }
97        };
98
99        // Add or update the project
100        if !user_data.projects.contains(&project_name.to_string()) {
101            user_data.projects.push(project_name.to_string());
102        }
103
104        // Update the map
105        self.user_projects
106            .insert(username.to_string(), user_data.clone());
107
108        // Save to database
109        let user_tree = self.db.open_tree(USER_DB_TREE)?;
110        let encoded = bincode::encode_to_vec(&user_data, bincode::config::standard())?;
111        user_tree.insert(username.as_bytes(), encoded)?;
112
113        Ok(())
114    }
115
116    /// Remove a project from a user's list
117    pub async fn remove_project(&self, username: &str, project_name: &str) -> Result<()> {
118        // Get user data
119        if let Some(mut user_data) = self.user_projects.get_mut(username) {
120            // Remove the project
121            user_data.projects.retain(|p| p != project_name);
122
123            // Save to database
124            let user_tree = self.db.open_tree(USER_DB_TREE)?;
125            let user_data_clone = user_data.clone();
126            let encoded = bincode::encode_to_vec(&user_data_clone, bincode::config::standard())?;
127            user_tree.insert(username.as_bytes(), encoded)?;
128        }
129
130        Ok(())
131    }

133    /// Verify that a user owns a function using the stored HMAC
134    pub fn verify_function_ownership(&self, username: &str, function_name: &str) -> bool {
135        if let Some(user_data) = self.user_projects.get(username) {
136            return user_data.projects.contains(&function_name.to_string());
137        }
138        false
139    }
