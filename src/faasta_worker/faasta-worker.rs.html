34fn main() {

37        process::exit(1);

41fn run() -> Result<()> {
42    let args = parse_args()?;
43    if let Some(parent) = args.ready_path.parent() {
44        fs::create_dir_all(parent)
45            .with_context(|| format!("failed to create ready dir {}", parent.display()))?;
46    }
47    let _ = fs::remove_file(&args.ready_path);
48
49    let library = unsafe {
50        Library::new(&args.artifact_path)
51            .with_context(|| format!("failed to load library {}", args.artifact_path.display()))?
52    };
53
54    let symbol_name = function_symbol_name(&args.function_name);
55    let handle_fn = unsafe {
56        let symbol: Symbol<HandleRequestFn> =
57            library.get(symbol_name.as_bytes()).with_context(|| {
58                format!(
59                    "function symbol '{symbol_name}' missing in {}",
60                    args.artifact_path.display()
61                )
62            })?;
63        *symbol
64    };
65
66    let region = SharedRegion::open(&args.shm_path)?;
67    fs::write(&args.ready_path, b"ready")
68        .with_context(|| format!("failed to write {}", args.ready_path.display()))?;
69
70    supervisor_loop(
71        region,
72        handle_fn,
73        &args.sandbox_path,
74        args.request_timeout_secs,
75    )
76}

78fn supervisor_loop(
79    region: SharedRegion,
80    handle_fn: HandleRequestFn,
81    sandbox_path: &Path,
82    request_timeout_secs: u32,
83) -> Result<()> {
84    let mut running = HashMap::<libc::pid_t, usize>::new();
85
86    loop {
87        reap_children(&region, &mut running);
88
89        let mut started = false;
90        for index in 0..region.slot_count() {
91            let slot = region.slot_header(index);
92            if slot
93                .state
94                .compare_exchange(
95                    STATE_READY,
96                    STATE_RUNNING,
97                    Ordering::AcqRel,
98                    Ordering::Acquire,
99                )
100                .is_err()
101            {
102                continue;
103            }
104
105            let pid = unsafe { libc::fork() };
106            match pid {
107                -1 => {
108                    region.write_error(index, "worker fork failed");
109                }
110                0 => child_main(region, index, handle_fn, sandbox_path, request_timeout_secs),
111                pid => {
112                    running.insert(pid, index);
113                    started = true;
114                }
115            }
116        }
117
118        if !started {
119            let seen = region.notify_value();
120            reap_children(&region, &mut running);
121            region.wait_for_work(seen);
122        }
123    }
124}

126fn child_main(
127    region: SharedRegion,
128    slot_index: usize,
129    handle_fn: HandleRequestFn,
130    sandbox_path: &Path,
131    request_timeout_secs: u32,
132) -> ! {
133    unsafe {
134        libc::alarm(request_timeout_secs);
135    }
136
137    let result = handle_request(&region, slot_index, handle_fn, sandbox_path)
138        .and_then(|response| region.write_response(slot_index, &response));
139    if let Err(err) = result {
140        region.write_error(slot_index, &err.to_string());
141    }
142    process::exit(0);
143}

145fn handle_request(
146    region: &SharedRegion,
147    slot_index: usize,
148    handle_fn: HandleRequestFn,
149    sandbox_path: &Path,
150) -> Result<WorkerResponse> {
151    let request = region.read_request(slot_index)?;
152    let runtime = tokio::runtime::Builder::new_current_thread()
153        .enable_time()
154        .enable_io()
155        .build()
156        .context("failed to build worker runtime")?;
157
158    runtime.block_on(async move {
159        let sandbox = Dir::open_ambient_dir(sandbox_path, ambient_authority())
160            .await
161            .with_context(|| format!("failed to open sandbox dir {}", sandbox_path.display()))?;
162        let future = unsafe { handle_fn(request.into_faasta(), sandbox) };
163        Ok::<_, anyhow::Error>(future.await.into())
164    })
165}

167fn reap_children(region: &SharedRegion, running: &mut HashMap<libc::pid_t, usize>) {
168    loop {
169        let mut status = 0;
170        let pid = unsafe { libc::waitpid(-1, &mut status, libc::WNOHANG) };
171        if pid <= 0 {
172            break;
173        }
174        if let Some(slot_index) = running.remove(&pid) {
175            let slot = region.slot_header(slot_index);
176            if slot.state.load(Ordering::Acquire) == STATE_RUNNING {
177                let message = child_status_message(status);
178                region.write_error(slot_index, &message);
179            }
180        }
181    }
182}
183
184fn child_status_message(status: i32) -> String {
185    if libc::WIFSIGNALED(status) {
186        format!(
187            "worker child terminated by signal {}",
188            libc::WTERMSIG(status)
189        )
190    } else if libc::WIFEXITED(status) {
191        format!(
192            "worker child exited with status {}",
193            libc::WEXITSTATUS(status)
194        )
195    } else {
196        "worker child exited before writing a response".to_string()
197    }
198}

200fn parse_args() -> Result<Args> {
201    let mut function_name = None;
202    let mut artifact_path = None;
203    let mut sandbox_path = None;
204    let mut shm_path = None;
205    let mut ready_path = None;
206    let mut request_timeout_secs = 10;
207
208    let mut args = std::env::args_os().skip(1);
209    while let Some(arg) = args.next() {
210        match arg.to_string_lossy().as_ref() {
211            "--function-name" => function_name = Some(next_string(&mut args, "--function-name")?),
212            "--artifact-path" => artifact_path = Some(next_path(&mut args, "--artifact-path")?),
213            "--sandbox-path" => sandbox_path = Some(next_path(&mut args, "--sandbox-path")?),
214            "--shm-path" => shm_path = Some(next_path(&mut args, "--shm-path")?),
215            "--ready-path" => ready_path = Some(next_path(&mut args, "--ready-path")?),
216            "--request-timeout-secs" => {
217                let value = next_string(&mut args, "--request-timeout-secs")?;
218                request_timeout_secs = value
219                    .parse()
220                    .with_context(|| format!("invalid --request-timeout-secs value: {value}"))?;
221            }
222            other => bail!("unknown argument: {other}"),
223        }
224    }
225
226    Ok(Args {
227        function_name: function_name.context("--function-name is required")?,
228        artifact_path: artifact_path.context("--artifact-path is required")?,
229        sandbox_path: sandbox_path.context("--sandbox-path is required")?,
230        shm_path: shm_path.context("--shm-path is required")?,
231        ready_path: ready_path.context("--ready-path is required")?,
232        request_timeout_secs,
233    })
234}

236fn next_string(args: &mut impl Iterator<Item = OsString>, name: &str) -> Result<String> {
237    args.next()
238        .with_context(|| format!("{name} requires a value"))?
239        .into_string()
240        .map_err(|_| anyhow::anyhow!("{name} must be valid UTF-8"))
241}
242
243fn next_path(args: &mut impl Iterator<Item = OsString>, name: &str) -> Result<PathBuf> {
244    Ok(PathBuf::from(next_string(args, name)?))
