28    pub async fn new(
29        metadata_db: Arc<Database>,
30        base_domain: String,
31        functions_dir: PathBuf,
32        invoker: FunctionInvoker,
33    ) -> Result<Self> {
34        if !functions_dir.exists() {
35            std::fs::create_dir_all(&functions_dir).with_context(|| {
36                format!(
37                    "failed to create functions directory at {:?}",
38                    functions_dir
39                )
40            })?;
41        }
42
43        let sandbox_root = functions_dir.join("sandbox");
44        std::fs::create_dir_all(&sandbox_root)
45            .with_context(|| format!("failed to create sandbox directory at {:?}", sandbox_root))?;
46
47        let github_auth = GitHubAuth::new(metadata_db.clone()).await?;
48
49        Ok(Self {
50            metadata_db,
51            base_domain,
52            functions_dir,
53            sandbox_root,
54            github_auth,
55            invoker,
56        })
57    }

59    pub fn artifact_path(&self, function_name: &str) -> PathBuf {
60        let wasm = self.functions_dir.join(format!("{function_name}.wasm"));
61        if wasm.exists() {
62            wasm
63        } else {
64            self.functions_dir.join(format!("{function_name}.cwasm"))
65        }
66    }

68    fn ensure_exists(path: &Path) -> Result<()> {
69        if !path.exists() {
70            bail!("function artifact missing at {}", path.display());
71        }
72        Ok(())
73    }

75    pub async fn prepare_sandbox_path(&self, function_name: &str) -> Result<PathBuf> {
76        let sandbox_path = self.sandbox_root.join(function_name);
77        std::fs::create_dir_all(&sandbox_path)
78            .with_context(|| format!("failed to prepare sandbox for {function_name}"))?;
79        Ok(sandbox_path)
80    }

82    pub async fn remove_from_cache(&self, function_name: &str) {
83        self.invoker.remove(function_name);
84        debug!("removed cached function runtime state {function_name}");
85    }
86
87    pub async fn invoke(
88        &self,
89        function_name: &str,
90        method: Method,
91        uri: Uri,
92        headers: HeaderMap,
93        body: Bytes,
94    ) -> Result<Response<Body>> {
95        let artifact_path = self.artifact_path(function_name);
96        Self::ensure_exists(&artifact_path)?;
97
98        let _sandbox_path = self
99            .prepare_sandbox_path(function_name)
100            .await
101            .with_context(|| format!("failed to prepare sandbox for '{function_name}'"))?;
102
103        let _timer = Timer::new(function_name.to_string());
104        let request = build_faasta_request(method, uri, headers, body);
105        let response = self
106            .invoker
107            .invoke(function_name, &artifact_path, request)
108            .await
109            .with_context(|| format!("worker failed for function '{function_name}'"))?;
110        Ok(faasta_response_to_http(response))
111    }
112
113    pub fn function_exists(&self, function_name: &str) -> bool {
114        self.artifact_path(function_name).exists()
115    }

145fn build_faasta_request(method: Method, uri: Uri, headers: HeaderMap, body: Bytes) -> WasmRequest {
146    let method_code = match method {
147        Method::GET => 0,
148        Method::POST => 1,
149        Method::PUT => 2,
150        Method::DELETE => 3,
151        Method::PATCH => 4,
152        Method::HEAD => 5,
153        Method::OPTIONS => 6,
154        _ => 0,
155    };
156
157    let mut header_vec = Vec::new();
158    for (name, value) in headers.iter() {
159        header_vec.push(WireHeader {
160            name: name.as_str().to_string(),
161            value: value.to_str().unwrap_or("").to_string(),
162        });
163    }
164
165    let uri_string = uri.to_string();
166
167    WasmRequest {
168        method: method_code,
169        uri: uri_string,
170        headers: header_vec,
171        body: body.to_vec(),
172    }
173}

175fn faasta_response_to_http(resp: WasmResponse) -> Response<Body> {
176    let mut response = Response::builder()
177        .status(resp.status)
178        .body(Body::from(resp.body))
179        .unwrap_or_else(|_| Response::builder().status(500).body(Body::empty()).unwrap());
180
181    let headers_mut = response.headers_mut();
182    for header in resp.headers {
183        if let (Ok(name), Ok(val)) = (
184            HeaderName::from_bytes(header.name.as_bytes()),
185            HeaderValue::from_str(header.value.as_str()),
186        ) {
187            headers_mut.append(name, val);
188        }
189    }
190
191    response
192}

194pub fn resolve_function_name(host: Option<&str>, path: &str, base_domain: &str) -> Option<String> {
195    if let Some(host) = host {
196        let host = host.split(':').next().unwrap_or(host);
197        if host.ends_with(base_domain) {
198            let parts = host.split('.').collect::<Vec<_>>();
199            if parts.len() > 2 {
200                let name = parts[0];
201                if !name.is_empty() {
202                    return Some(name.to_string());
203                }
204            }
205        }
206    }
207
208    let trimmed = path.trim_start_matches('/');
209    if trimmed.is_empty() {
210        None
211    } else {
212        trimmed.split('/').next().map(|s| s.to_string())
213    }
214}

216pub fn sanitize_function_name(function_name: &str) -> Option<String> {
217    let valid = function_name
218        .chars()
219        .all(|c| c.is_ascii_alphanumeric() || c == '-' || c == '_');
220    if valid {
221        Some(function_name.to_string())
222    } else {
223        None
224    }
