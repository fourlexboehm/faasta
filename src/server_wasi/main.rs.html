53// Create a basic response with string content
54fn text_response(status: u16, text: &str) -> Result<Response<HyperOutgoingBody>> {
55    // Create a simple body with the provided text
56    // Clone the text to ensure it's owned data that will live beyond this function
57    let text_owned = text.to_string();
58    let body = Full::new(Bytes::from(text_owned))
59        .map_err(|_| ErrorCode::InternalError(None))
60        .boxed();
61
62    // Build and return the response
63    Ok(Response::builder()
64        .status(status)
65        .header("Content-Type", "text/plain")
66        .body(HyperOutgoingBody::new(body))?)
67}

69// Create a redirect response to website.faasta.xyz
70fn redirect_to_website() -> Result<Response<HyperOutgoingBody>> {
71    // Create a simple body with a redirect message
72    let text_owned = "Redirecting to website.faasta.xyz...".to_string();
73    let body = Full::new(Bytes::from(text_owned))
74        .map_err(|_| ErrorCode::InternalError(None))
75        .boxed();
76
77    // Build and return the redirect response
78    Ok(Response::builder()
79        .status(302)
80        .header("Location", "https://website.faasta.xyz")
81        .header("Content-Type", "text/plain")
82        .body(HyperOutgoingBody::new(body))?)
83}

150    async fn handle_request(&self, req: Request<Incoming>) -> Result<Response<HyperOutgoingBody>> {
151        // Extract function name from subdomain or path
152        let host_header = req
153            .headers()
154            .get(HOST)
155            .and_then(|h| h.to_str().ok())
156            .map(|s| s.to_string());
157        let path = req.uri().path().to_string();
158
159        debug!("Handling request with path: {}", path);
160
161        // Check if it's the root domain or local development host
162        if host_header
163            .as_deref()
164            .map(|h| {
165                h == self.base_domain || h.starts_with("localhost") || h.starts_with("127.0.0.1")
166            })
167            .unwrap_or(false)
168        {
169            debug!("Processing request on root domain: {}", self.base_domain);
170            // Root domain with no subdomain - try to route based on path
171            let path_str = path.as_str();
172            let path_parts: Vec<&str> = path_str.split('/').collect();
173
174            if path_parts.len() >= 2 && !path_parts[1].is_empty() {
175                let function_name = path_parts[1].to_string();
176                debug!(
177                    "Processing path-based request for function: {}",
178                    function_name
179                );
180
181                // Use direct function name approach
182                let wasm_filename = format!("{}.wasm", function_name);
183                debug!("Looking for WASM file: {}", wasm_filename);
184
185                // Create a timer for this function call
186                let _timer = Timer::new(function_name.clone());
187                let function_path = self.functions_dir.join(&wasm_filename);
188
189                // Debug logging to track function path
190                if function_path.exists() {
191                    debug!("Found function at path: {:?}", function_path);
192                    // Create a new path to remove the /{function_name} prefix
193                    let new_path = if path_parts.len() > 2 {
194                        // Keep the rest of the path
195                        format!("/{}", path_parts[2..].join("/"))
196                    } else {
197                        // Just the root
198                        "/".to_string()
199                    };
200
201                    debug!("Rewriting path to: {}", new_path);
202
203                    // Build a new request with the modified path
204                    let mut builder = Request::builder()
205                        .method(req.method().clone())
206                        .uri(new_path)
207                        .version(req.version());
208
209                    // Copy all headers
210                    for (name, value) in req.headers() {
211                        builder = builder.header(name, value);
212                    }
213
214                    let (_, body) = req.into_parts();
215                    let new_req = builder.body(body)?;
216
217                    return self
218                        .execute_function(new_req, &function_name, &function_path)
219                        .await;
220                } else {
221                    debug!("Function not found at path: {:?}", function_path);
222                    // If we're looking for a specific function but it doesn't exist, return a 404
223                    return text_response(404, &format!("Function '{}' not found", function_name));
224                }
225            }
226
227            // No function found in path, redirect to website
228            debug!("No function specified in path, redirecting to website");
229            return redirect_to_website();
230        }
231
232        if let Some(host) = &host_header {
233            let expected_suffix = format!(".{}", self.base_domain);
234            debug!("Checking host: {} for subdomain routing", host);
235
236            if !host
237                .to_lowercase()
238                .ends_with(&expected_suffix.to_lowercase())
239                && !host.starts_with("localhost")
240                && !host.starts_with("127.0.0.1")
241            {
242                debug!(
243                    "Host doesn't end with expected suffix: {} and is not a local development host",
244                    expected_suffix
245                );
246                return redirect_to_website();
247            }
248
249            let subdomain = host.trim_end_matches(&expected_suffix).to_string();
250            if subdomain.is_empty() || subdomain == *host {
251                debug!("Empty subdomain or hostname equals subdomain, redirecting");
252                return redirect_to_website();
253            }
254
255            debug!("Processing subdomain request for function: {}", subdomain);
256
257            // Use direct function name approach
258            let wasm_filename = format!("{}.wasm", subdomain);
259            debug!("Looking for WASM file: {}", wasm_filename);
260
261            // Create a timer for this function call
262            let _timer = Timer::new(subdomain.clone());
263            let function_path = self.functions_dir.join(&wasm_filename);
264            if !function_path.exists() {
265                debug!("Function not found at path: {:?}", function_path);
266                return text_response(404, &format!("Function '{}' not found", subdomain));
267            }
268
269            // Execute the function
270            debug!("Executing function from subdomain route");
271            return self.execute_function(req, &subdomain, &function_path).await;
272        } else {
273            // No host header, redirect to website
274            debug!("No host header found, redirecting to website");
275            redirect_to_website()
276        }
277    }

279    async fn execute_function(
280        &self,
281        req: Request<Incoming>,
282        function_name: &str,
283        function_path: &PathBuf,
284    ) -> Result<Response<HyperOutgoingBody>> {
285        debug!(
286            "Executing function: {} with path: {:?}",
287            function_name, function_path
288        );
289        // Get or load the ProxyPre
290        let pre = self
291            .get_or_load_proxy_pre(function_name, function_path)
292            .await?;
293
294        // Create store with client state
295        let mut store = Store::new(
296            pre.engine(),
297            FaastaClientState {
298                table: ResourceTable::new(),
299                wasi: WasiCtxBuilder::new()
300                    .inherit_stdio()
301                    .env("FUNCTION_NAME", function_name)
302                    .build(),
303                http: WasiHttpCtx::new(),
304                function_name: function_name.to_string(),
305            },
306        );
307
308        // Setup the response channel
309        let (sender, receiver) = tokio::sync::oneshot::channel();
310
311        // Create the WASI HTTP request
312        let wasi_req = store.data_mut().new_incoming_request(Scheme::Http, req)?;
313        let wasi_resp_out = store.data_mut().new_response_outparam(sender)?;
314
315        // Clone the pre so we can move it into the task
316        let pre_clone = pre.clone();
317
318        // Spawn task to execute the function
319        let task = tokio::task::spawn(async move {
320            let proxy = pre_clone.instantiate_async(&mut store).await?;
321            proxy
322                .wasi_http_incoming_handler()
323                .call_handle(store, wasi_req, wasi_resp_out)
324                .await?;
325            Ok::<_, anyhow::Error>(())
326        });
327
328        // Wait for response with a 10-minute timeout
329        match tokio::time::timeout(std::time::Duration::from_secs(600), receiver).await {
330            Ok(receiver_result) => match receiver_result {
331                Ok(Ok(resp)) => Ok(resp),
332                Ok(Err(err_code)) => {
333                    error!("Function returned error: {:?}", err_code);
334                    Err(anyhow!("Function error: {:?}", err_code))
335                }
336                Err(_) => match task.await {
337                    Ok(Ok(())) => bail!("Function did not set response"),
338                    Ok(Err(e)) => Err(e),
339                    Err(e) => Err(e.into()),
340                },
341            },
342            Err(_) => {
343                error!("Function execution timed out after 10 minutes");
344                Err(anyhow!("Function execution timed out after 10 minutes"))
345            }
346        }
347    }

349    async fn get_or_load_proxy_pre(
350        &self,
351        function_name: &str,
352        function_path: &PathBuf,
353    ) -> Result<ProxyPre<FaastaClientState>> {
354        // Check if we have this pre-cached
355        if let Some(cached) = self.pre_cache.get(function_name) {
356            return Ok(cached.value().clone());
357        }
358
359        // Load the component and create a linker
360        let component = Component::from_file(&self.engine, function_path)?;
361        let mut linker = Linker::new(&self.engine);
362
363        // wasmtime_wasi::add_to_linker_async(&mut linker)?;
364        // let link_options = self.run.compute_wasi_features();
365        let options = LinkOptions::default();
366        wasmtime_wasi::add_to_linker_with_options_async(&mut linker, &options)?;
367        wasmtime_wasi_http::add_only_http_to_linker_async(&mut linker)?;
368        // wasmtime_wasi::add_to_linker_async(&mut linker)?;
369        // wasmtime_wasi::bindings::cli::environment::add_to_linker_get_host(&mut linker, |_| {})?;
370
371        // Create the ProxyPre
372        let pre = ProxyPre::new(linker.instantiate_pre(&component)?)?;
373
374        // Cache it for future use
375        self.pre_cache
376            .insert(function_name.to_string(), pre.clone());
377
378        Ok(pre)
379    }

433async fn load_tls_config(args: &Args) -> Result<Arc<ServerConfig>> {
434    // Load TLS certificate
435    let cert_file = File::open(&args.tls_cert_path)
436        .with_context(|| format!("Failed to open TLS cert file: {:?}", args.tls_cert_path))?;
437    let mut cert_reader = BufReader::new(cert_file);
438    let certs = certs(&mut cert_reader)
439        .collect::<Result<Vec<_>, _>>()
440        .context("Failed to parse TLS certificates")?;
441
442    // Load TLS private key
443    let key_file = File::open(&args.tls_key_path)
444        .with_context(|| format!("Failed to open TLS key file: {:?}", args.tls_key_path))?;
445    let mut key_reader = BufReader::new(key_file);
446    let key = private_key(&mut key_reader)?.context("No private key found in TLS key file")?;
447
448    // Build TLS config
449    let config = ServerConfig::builder()
450        .with_no_client_auth()
451        .with_single_cert(certs, key)
452        .context("Failed to build TLS server config")?;
453
454    Ok(Arc::new(config))
455}

457// Function to handle connections for tarpc
458async fn run_server(mut quic_server: s2n_quic::Server, server: Arc<FaastaServer>) {
459    while let Some(mut connection) = quic_server.accept().await {
460        let server_clone = server.clone();
461        tokio::spawn(async move {
462            debug!("Accepted new connection");
463
464            while let Ok(Some(stream)) = connection.accept_bidirectional_stream().await {
465                let server_clone = server_clone.clone();
466                tokio::spawn(async move {
467                    debug!("Accepted new stream");
468                    let framed = LengthDelimitedCodec::builder().new_framed(stream);
469                    let transport = transport::new(framed, Bincode::default());
470
471                    // Create a cloned service for this connection
472                    let github_auth_clone = server_clone.github_auth.clone();
473                    let service_clone = rpc_service::create_service_with_github_auth(
474                        server_clone.functions_dir.clone(),
475                        github_auth_clone,
476                        server_clone.metadata_db.clone(),
477                    )
478                    .expect("Failed to create function service");
479
480                    // Process this connection
481                    // Use default configuration but with a longer context deadline
482                    let server_channel = BaseChannel::with_defaults(transport);
483                    server_channel
484                        .execute(service_clone.serve())
485                        .for_each(|fut| {
486                            tokio::spawn(fut);
487                            async {}
488                        })
489                        .await;
490                });
491            }
492        });
493    }
494}

496// HTTP to HTTPS redirection using Axum framework
497async fn run_http_redirect_server(http_listener: TcpListener) {
498    info!("HTTP redirect server listening for connections");
499
500    // Create a function to convert HTTP URLs to HTTPS
501    let make_https = |host: &str, uri: Uri, https_port: u16| -> Result<Uri, BoxError> {
502        let mut parts = uri.into_parts();
503
504        parts.scheme = Some(axum::http::uri::Scheme::HTTPS);
505
506        if parts.path_and_query.is_none() {
507            parts.path_and_query = Some("/".parse().unwrap());
508        }
509
510        let authority: Authority = host.parse()?;
511        let bare_host = match authority.port() {
512            Some(port_struct) => authority
513                .as_str()
514                .strip_suffix(port_struct.as_str())
515                .unwrap()
516                .strip_suffix(':')
517                .unwrap(), // if authority.port() is Some(port) then we can be sure authority ends with :{port}
518            None => authority.as_str(),
519        };
520
521        parts.authority = Some(format!("{bare_host}:{https_port}").parse()?);
522
523        Ok(Uri::from_parts(parts)?)
524    };
525
526    // Get the local port this listener is bound to
527    let listener_addr = http_listener.local_addr().unwrap();
528
529    // Determine HTTPS port (default to 443)
530    let https_port = 443;
531
532    // Create the redirect handler
533    let redirect = move |Host(host): Host, uri: Uri| async move {
534        match make_https(&host, uri, https_port) {
535            Ok(uri) => Ok(Redirect::permanent(&uri.to_string())),
536            Err(error) => {
537                tracing::warn!(%error, "failed to convert URI to HTTPS");
538                Err(axum::http::StatusCode::BAD_REQUEST)
539            }
540        }
541    };
542
543    // Create Axum router with the redirect handler
544    let app = Router::new().fallback(redirect);
545
546    // Start the Axum HTTP server
547    info!(
548        "HTTP redirect service listening on http://{}",
549        listener_addr
550    );
551
552    // Serve with the existing TcpListener
553    axum::serve(http_listener, app).await.unwrap();
554}

556#[tokio::main]
557async fn main() -> Result<()> {
558    // Install default crypto provider for rustls
559    rustls::crypto::ring::default_provider()
560        .install_default()
561        .expect("Failed to install crypto provider");
562
563    // Load environment variables from .env file if present
564    let _ = dotenvy::dotenv();
565
566    // Initialize tracing
567    tracing_subscriber::fmt().with_max_level(Level::INFO).init();
568
569    // Parse command-line arguments
570    let args = Args::parse();
571
572    // Ensure required directories exist
573    std::fs::create_dir_all(&args.db_path)?;
574    std::fs::create_dir_all(&args.functions_path)?;
575    std::fs::create_dir_all(&args.certs_dir)?;
576
577    // Setup certificate management
578    if args.auto_cert {
579        // Create CertManager instance for Porkbun
580        let cert_manager = CertManager::new(
581            args.base_domain.clone(),
582            args.certs_dir.clone(),
583            args.tls_cert_path.clone(),
584            args.tls_key_path.clone(),
585        );
586
587        // Obtain or renew certificate if needed
588        cert_manager
589            .obtain_or_renew_certificate()
590            .await
591            .context("Failed to obtain/renew TLS certificate")?;
592    }
593
594    // Create a clone for use in the QUIC server task
595    let args_clone = args.clone();
596
597    info!(
598        "Starting server-wasi with base domain: {}",
599        args.base_domain
600    );
601
602    // Ensure metrics database directory exists
603    let metrics_db_path =
604        std::env::var("METRICS_DB_PATH").unwrap_or_else(|_| "./data/metrics".to_string());
605    std::fs::create_dir_all(
606        std::path::Path::new(&metrics_db_path)
607            .parent()
608            .unwrap_or_else(|| std::path::Path::new(".")),
609    )?;
610
611    // Open/create component cache database
612    let pre_cache = sled::open(&args.db_path)?;
613
614    // Configure Wasmtime engine
615
616    let mut config = Config::new();
617    config.async_support(true);
618    config.wasm_component_model(true);
619
620    // Create the engine
621    let engine = Engine::new(&config)?;
622
623    // Create server
624    let server_instance = Arc::new(
625        FaastaServer::new(
626            engine,
627            pre_cache,
628            args.base_domain.clone(),
629            args.functions_path.clone(),
630        )
631        .await?,
632    );
633
634    // Store server in global OnceCell for cache management
635    let _ = SERVER.set(server_instance.clone());
636
637    // Spawn a background task to flush metrics to DB every hour
638    metrics::spawn_periodic_flush(60 * 30);
639
640    // Load TLS configuration
641    let tls_config = load_tls_config(&args).await?;
642
643    let tls_acceptor = TlsAcceptor::from(tls_config.clone());
644
645    // Start listening for HTTP connections (for redirects)
646    let http_listener = TcpListener::bind(&args.http_listen_addr)
647        .await
648        .with_context(|| format!("Failed to bind to {}", args.http_listen_addr))?;
649    info!(
650        "HTTP redirect service listening on http://{}",
651        args.http_listen_addr
652    );
653
654    // Start HTTP redirect server as a separate tokio task
655    tokio::spawn(run_http_redirect_server(http_listener));
656
657    // Start listening for HTTPS connections
658    let listener = TcpListener::bind(&args.listen_addr)
659        .await
660        .with_context(|| format!("Failed to bind to {}", args.listen_addr))?;
661    info!("Listening on https://{}", args.listen_addr);
662
663    // Start tarpc service for function management
664    let server_clone = server_instance.clone();
665    tokio::spawn(async move {
666        let addr = "0.0.0.0:4433".parse::<std::net::SocketAddr>().unwrap();
667
668        // Configure server with the TLS certs
669        let quic_server = s2n_quic::Server::builder()
670            .with_tls((
671                Path::new(&args_clone.tls_cert_path),
672                Path::new(&args_clone.tls_key_path),
673            ))
674            .map_err(|e| anyhow::anyhow!("Failed to set up TLS: {:?}", e))
675            .expect("Failed to set up TLS")
676            .with_io(addr)
677            .map_err(|e| anyhow::anyhow!("Failed to set up IO: {:?}", e))
678            .expect("Failed to set up IO")
679            .start()
680            .map_err(|e| anyhow::anyhow!("Failed to start server: {:?}", e))
681            .expect("Failed to start server");
682
683        info!("RPC service listening on {}", addr);
684
685        // Process connections
686        run_server(quic_server, server_clone).await;
687    });
688    // Main server loop
689    loop {
690        // Accept incoming connection
691        let (stream, peer_addr) = match listener.accept().await {
692            Ok(conn) => conn,
693            Err(e) => {
694                error!("Failed to accept connection: {}", e);
695                continue;
696            }
697        };
698        info!("Accepted connection from {}", peer_addr);
699
700        // Clone server and acceptor for this connection
701        let server = server_instance.clone();
702        let tls_acceptor = tls_acceptor.clone();
703
704        // Handle connection in a new task
705        tokio::spawn(async move {
706            // Perform TLS handshake
707            match tls_acceptor.accept(stream).await {
708                Ok(tls_stream) => {
709                    info!("TLS handshake successful with {}", peer_addr);
710
711                    // Create a service function for handling HTTP requests
712                    let service = service_fn(move |req: Request<Incoming>| {
713                        let server = server.clone();
714                        async move {
715                            match server.handle_request(req).await {
716                                Ok(response) => Ok::<_, anyhow::Error>(response),
717                                Err(e) => {
718                                    error!("Error handling request: {}", e);
719                                    // Return a generic 500 error response
720                                    match text_response(500, "Internal Server Error") {
721                                        Ok(resp) => Ok(resp),
722                                        Err(err) => {
723                                            error!("Failed to create error response: {}", err);
724                                            // Fall back to a minimal hard-coded response if everything else fails
725                                            let error_text = "Internal Server Error".to_string();
726                                            let body = Full::new(Bytes::from(error_text))
727                                                .map_err(|_| ErrorCode::InternalError(None))
728                                                .boxed();
729                                            Ok(Response::builder()
730                                                .status(500)
731                                                .header("Content-Type", "text/plain")
732                                                .body(HyperOutgoingBody::new(body))
733                                                .unwrap())
734                                        }
735                                    }
736                                }
737                            }
738                        }
739                    });
740
741                    // Serve the HTTP connection directly with hyper
742                    if let Err(err) = http1::Builder::new()
743                        .serve_connection(TokioIo::new(tls_stream), service)
744                        .await
745                    {
746                        // Only log errors that aren't from client disconnects
747                        if !err.is_closed() && !err.is_canceled() {
748                            error!("Error serving connection from {}: {}", peer_addr, err);
749                        }
750                    }
751                }
752                Err(e) => {
753                    error!("TLS handshake failed with {}: {}", peer_addr, e);
754                }
755            }
756        });
757    }
