83async fn load_tls_config(args: &Args) -> Result<Arc<ServerConfig>> {
84    // Load TLS certificate
85    let cert_file = File::open(&args.tls_cert_path)
86        .with_context(|| format!("Failed to open TLS cert file: {:?}", args.tls_cert_path))?;
87    let mut cert_reader = BufReader::new(cert_file);
88
89    // Parse the certificates
90    let certs = rustls_pemfile::certs(&mut cert_reader).collect::<Result<Vec<_>, _>>()?;
91
92    // Load TLS private key
93    let key_file = File::open(&args.tls_key_path)
94        .with_context(|| format!("Failed to open TLS key file: {:?}", args.tls_key_path))?;
95    let mut key_reader = BufReader::new(key_file);
96
97    // Parse the private key
98    let key = rustls_pemfile::private_key(&mut key_reader)?
99        .ok_or_else(|| anyhow::anyhow!("No private key found in TLS key file"))?;
100
101    // Build TLS config
102    let config = ServerConfig::builder()
103        .with_no_client_auth()
104        .with_single_cert(certs, key)
105        .context("Failed to build TLS server config")?;
106
107    Ok(Arc::new(config))
108}

113#[compio::main]
114async fn main() -> anyhow::Result<()> {
115    // Install default crypto provider for rustls
116    rustls::crypto::ring::default_provider()
117        .install_default()
118        .expect("Failed to install crypto provider");
119
120    // Load environment variables from .env file if present
121    let _ = dotenvy::dotenv();
122
123    // Initialize tracing
124    tracing_subscriber::fmt().with_max_level(Level::INFO).init();
125
126    // Parse command-line arguments
127    let args = Args::parse();
128
129    // Ensure required directories exist
130    std::fs::create_dir_all(&args.db_path)?;
131    std::fs::create_dir_all(&args.functions_path)?;
132    std::fs::create_dir_all(&args.certs_dir)?;
133
134    // Setup certificate management
135    if args.auto_cert {
136        // Create CertManager instance for Porkbun
137        let cert_manager = Arc::new(CertManager::new(
138            args.base_domain.clone(),
139            args.certs_dir.clone(),
140            args.tls_cert_path.clone(),
141            args.tls_key_path.clone(),
142        ));
143
144        cert_manager
145            .obtain_or_renew_certificate()
146            .await
147            .context("Failed to obtain/renew TLS certificate")?;
148
149        // Spawn periodic certificate download (every 7 days)
150        info!("Starting periodic certificate downloads every 7 days");
151        cert_manager.spawn_periodic_renewal();
152    }
153
154    // Pre-compile available functions to improve startup time
155    async fn precompile_functions(engine: &Engine, functions_dir: &Path) -> Result<()> {
156        info!("Pre-compiling functions...");
157
158        let function_files = std::fs::read_dir(functions_dir)?
159            .filter_map(|entry| {
160                let entry = entry.ok()?;
161                let path = entry.path();
162                if path.extension().and_then(|ext| ext.to_str()) == Some("wasm") {
163                    Some(path)
164                } else {
165                    None
166                }
167            })
168            .collect::<Vec<_>>();
169
170        // Log how many functions we're going to precompile
171        info!("Found {} functions to precompile", function_files.len());
172
173        // Precompile each function
174        for path in function_files {
175            let filename = path.file_name().unwrap().to_string_lossy();
176            info!("Precompiling function: {}", filename);
177            let wasm = fs::read(&path).unwrap();
178            let cwasm = engine.precompile_component(&wasm).unwrap();
179            fs::write(path.with_extension("cwasm"), cwasm).unwrap();
180        }
181
182        info!("Precompilation complete");
183        Ok(())
184    }
185
186    info!(
187        "Starting server-wasi with base domain: {}",
188        args.base_domain
189    );
190
191    // Ensure metrics database directory exists
192    let metrics_db_path =
193        std::env::var("METRICS_DB_PATH").unwrap_or_else(|_| "./data/metrics".to_string());
194    std::fs::create_dir_all(
195        std::path::Path::new(&metrics_db_path)
196            .parent()
197            .unwrap_or_else(|| std::path::Path::new(".")),
198    )?;
199
200    // Open/create component cache database
201    let metadata_db = sled::open(&args.db_path)?;
202    let mut config = Config::default();
203    config.async_support(true);
204    config.wasm_component_model(true);
205    config.memory_init_cow(true);
206    let mut pool = PoolingAllocationConfig::new();
207    pool.total_memories(100);
208    pool.max_memory_size(1 << 28); // 256 MiB
209    pool.total_tables(100);
210    pool.table_elements(5000);
211    pool.total_core_instances(100);
212    config.allocation_strategy(InstanceAllocationStrategy::Pooling(pool));
213
214    // Enable module caching to speed up startup time
215    config.cache(Some(Cache::from_file(None)?));
216
217    config.cranelift_opt_level(OptLevel::Speed);
218
219    // Enable parallel compilation if available
220    config.parallel_compilation(true);
221
222    // Precompile modules ahead of time
223    config.strategy(wasmtime::Strategy::Cranelift);
224
225    // Create the engine
226    let engine = Engine::new(&config)?;
227
228    // // Precompile functions
229    if let Err(e) = precompile_functions(&engine, &args.functions_path).await {
230        error!("Error precompiling functions: {}", e);
231    }
232
233    // Create server
234    let server_instance = wasi_server::FaastaServer::new(
235        engine,
236        metadata_db,
237        args.base_domain.clone(),
238        args.functions_path.clone(),
239    )
240    .await?;
241
242    // Store server in global OnceCell for cache management
243    let _ = SERVER.set(server_instance);
244
245    // Spawn a background task to flush metrics to DB
246    metrics::spawn_periodic_flush(60 * 30);
247
248    // Load TLS configuration with timing
249    info!("Loading TLS configuration...");
250    let tls_timer = metrics::Timer::new("tls_config_loading".to_string());
251    let tls_config = load_tls_config(&args).await?;
252    drop(tls_timer); // Explicitly drop to record timing
253    info!("TLS configuration loaded");
254
255    let tls_acceptor = TlsAcceptor::from(tls_config.clone());
256
257    let worker_threads = NonZeroUsize::new(num_cpus::get_physical())
258        .unwrap_or_else(|| NonZeroUsize::new(1).unwrap());
259    let dispatcher = Arc::new(
260        Dispatcher::builder()
261            .worker_threads(worker_threads)
262            .build()
263            .context("failed to build HTTP dispatcher")?,
264    );
265
266    let http_listener = TcpListener::bind(&args.http_listen_addr)
267        .await
268        .with_context(|| format!("Failed to bind to {}", args.http_listen_addr))?;
269    {
270        let dispatcher = dispatcher.clone();
271        spawn(async move {
272            http::run_http_server(http_listener, dispatcher).await;
273        })
274        .detach();
275    }
276
277    // Start listening for HTTPS connections
278    let listener = TcpListener::bind(&args.listen_addr)
279        .await
280        .with_context(|| format!("Failed to bind to {}", args.listen_addr))?;
281    info!("Listening on https://{}", args.listen_addr);
282
283    // Start RPC service for function management on a dedicated compio runtime
284    let rpc_address = String::from("0.0.0.0:4433");
285    {
286        let tls_cert_path = args.tls_cert_path.clone();
287        let tls_key_path = args.tls_key_path.clone();
288        spawn(async move {
289            if let Err(e) = quic::run_rpc_server(tls_cert_path, tls_key_path, rpc_address).await {
290                error!("RPC server terminated: {e}");
291            }
292        })
293        .detach();
294    }
295
296    // Run HTTPS server in the main thread
297    http::run_https_server(listener, tls_acceptor, dispatcher.clone()).await;
298    Ok(())
299}
