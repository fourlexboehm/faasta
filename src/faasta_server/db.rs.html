11    pub fn open(base_path: &Path) -> Result<Self> {
12        let db_path = sqlite_path(base_path, "faasta.sqlite3");
13        if let Some(parent) = db_path.parent() {
14            std::fs::create_dir_all(parent)
15                .with_context(|| format!("failed to create sqlite parent dir {:?}", parent))?;
16        }
17
18        let conn =
19            Connection::open(&db_path).with_context(|| format!("failed to open {:?}", db_path))?;
20        conn.pragma_update(None, "journal_mode", "WAL")?;
21        conn.pragma_update(None, "synchronous", "NORMAL")?;
22        conn.pragma_update(None, "foreign_keys", "ON")?;
23
24        let db = Self {
25            conn: Mutex::new(conn),
26        };
27        db.init_schema()?;
28        Ok(db)
29    }

31    fn init_schema(&self) -> Result<()> {
32        let conn = self.conn.lock().expect("sqlite mutex poisoned");
33        conn.execute_batch(
34            "CREATE TABLE IF NOT EXISTS functions (
35                name TEXT PRIMARY KEY,
36                data BLOB NOT NULL
37            );
38            CREATE TABLE IF NOT EXISTS user_data (
39                username TEXT PRIMARY KEY,
40                data BLOB NOT NULL
41            );
42            CREATE TABLE IF NOT EXISTS metrics (
43                function_name TEXT PRIMARY KEY,
44                total_time INTEGER NOT NULL,
45                call_count INTEGER NOT NULL,
46                last_called INTEGER NOT NULL
47            );",
48        )?;
49        Ok(())
50    }

160fn sqlite_path(base_path: &Path, default_name: &str) -> PathBuf {
161    if base_path.extension().is_some() {
162        base_path.to_path_buf()
163    } else {
164        base_path.join(default_name)
165    }
