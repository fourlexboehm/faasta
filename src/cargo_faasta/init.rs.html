6pub const HTTP_CARGO_TOML: &str = include_str!("../../function/Cargo.toml");
7pub const HTTP_LIB_RS: &str = include_str!("../../function/src/lib.rs");
8pub fn handle_new(args: &NewArgs) -> Result<(), Box<dyn Error>> {
9    dbg!(&args);
10    let current_dir = env::current_dir()?;
11    let new_project_dir = current_dir.join(&args.package_name);
12
13    if new_project_dir.exists() && !args.package_name.is_empty() {
14        return Err(format!("Directory '{}' already exists", args.package_name).into());
15    }
16    if new_project_dir.join("Cargo.toml").exists() {
17        return Err(format!(
18            "Cargo.toml already exists in '{}'",
19            new_project_dir.display()
20        )
21        .into());
22    }
23    fs::create_dir_all(new_project_dir.join("src"))?;
24    let pkg_name = if args.package_name.is_empty() {
25        "axum_serverless"
26    } else {
27        &*args.package_name
28    };
29
30    write_files(&new_project_dir, HTTP_CARGO_TOML, HTTP_LIB_RS, pkg_name)?;
31
32    println!(
33        "Successfully created new Axum project '{}' at '{}'",
34        args.package_name,
35        new_project_dir.display()
36    );
37    Ok(())
38}

40/// Writes the embedded Cargo.toml & main.rs to disk,
41/// updating the `[package] name` in Cargo.toml to `package_name`.
42fn write_files(
43    project_dir: &Path,
44    cargo_toml_str: &str,
45    main_rs_str: &str,
46    package_name: &str,
47) -> io::Result<()> {
48    // 1. Write Cargo.toml
49    // EVENT_PATHS
50
51    let cargo_toml_path = project_dir.join("Cargo.toml");
52    let updated_cargo_toml = rewrite_package_name(cargo_toml_str, package_name);
53    fs::write(cargo_toml_path, updated_cargo_toml)?;
54
55    // 2. Write src/lib.rs
56    let main_rs_path = project_dir.join("src").join("lib.rs");
57    fs::write(main_rs_path, main_rs_str)?;
58
59    Ok(())
60}

62/// Replaces the line `name = "whatever"` inside `[package]` with the user-provided `package_name`.
63fn rewrite_package_name(toml_input: &str, package_name: &str) -> String {
64    let mut in_package = false;
65    let mut output = String::new();
66
67    for line in toml_input.lines() {
68        let trimmed = line.trim_start();
69        if trimmed.starts_with("[package]") {
70            in_package = true;
71            output.push_str(line);
72            output.push('\n');
73            continue;
74        }
75
76        if in_package && trimmed.starts_with("name =") {
77            output.push_str(&format!("name = \"{}\"\n", package_name));
78            in_package = false; // Only replace once
79            continue;
80        }
81
82        output.push_str(line);
83        output.push('\n');
84    }
85
86    output
87}
