58/// Performs the GitHub OAuth flow and returns the username and token
59pub async fn github_oauth_flow() -> Result<(String, String)> {
60    // Check if we're in test mode
61    let (is_test_mode, test_username, test_token) = get_test_data();
62    if is_test_mode {
63        if let (Some(username), Some(token)) = (test_username, test_token) {
64            println!("Using test credentials");
65            return Ok((username, format!("Bearer {}", token)));
66        }
67    }
68
69    // Set up the OAuth2 client
70    let github_client = get_oauth_client()?;
71
72    // Generate the authorization URL
73    let (authorize_url, csrf_state) = github_client
74        .authorize_url(CsrfToken::new_random)
75        .add_scope(Scope::new("user:email".to_string()))
76        .url();
77
78    // Start the redirect server
79    let server = start_redirect_server()?;
80
81    // Open the browser to authenticate the user
82    println!("Opening browser for GitHub authentication...");
83    println!("Authorization URL: {}", authorize_url);
84    if let Err(e) = open::that(authorize_url.to_string()) {
85        return Err(anyhow!("Failed to open browser: {}", e));
86    }
87
88    // Wait for the callback from GitHub
89    println!("Waiting for GitHub authentication...");
90    let auth_code = wait_for_callback(server, &csrf_state)?;
91
92    // Exchange the authorization code for a token
93    println!("Exchanging authorization code for token...");
94    let token = match github_client
95        .exchange_code(AuthorizationCode::new(auth_code))
96        .request_async(async_http_client)
97        .await
98    {
99        Ok(token) => token,
100        Err(e) => {
101            println!("Error exchanging code for token: {:?}", e);
102            return Err(anyhow!(
103                "Failed to exchange authorization code for token: {}",
104                e
105            ));
106        }
107    };
108
109    // Get the access token as a string
110    let access_token = token.access_token().secret();
111
112    // Get the user's GitHub info using the token
113    println!("Getting GitHub user information...");
114    let username = get_github_username(access_token).await?;
115
116    Ok((username, format!("Bearer {}", access_token)))
117}

119/// Create an OAuth client for GitHub
120fn get_oauth_client() -> Result<BasicClient> {
121    let redirect_url = format!("http://localhost:{}/oauth/callback", REDIRECT_PORT);
122    println!("Redirect URL: {}", redirect_url);
123
124    Ok(BasicClient::new(
125        ClientId::new(get_client_id()),
126        Some(ClientSecret::new(get_client_secret())),
127        AuthUrl::new("https://github.com/login/oauth/authorize".to_string())?,
128        Some(TokenUrl::new(
129            "https://github.com/login/oauth/access_token".to_string(),
130        )?),
131    )
132    .set_redirect_uri(RedirectUrl::new(redirect_url)?))
133}

135/// Starts a local HTTP server to receive the OAuth redirect
136fn start_redirect_server() -> Result<Server> {
137    let addr = SocketAddr::from_str(&format!("127.0.0.1:{}", REDIRECT_PORT))?;
138    let server = Server::http(addr).map_err(|e| anyhow!("Failed to start server: {}", e))?;
139    Ok(server)
140}

142/// Waits for and processes the OAuth callback
143fn wait_for_callback(server: Server, csrf_state: &CsrfToken) -> Result<String> {
144    // Wait for the callback from GitHub
145    let req = server.recv()?;
146
147    // Parse the request URL to extract the code and state
148    let url_str = format!("http://localhost{}", req.url());
149    let url = Url::parse(&url_str)?;
150
151    // Extract query parameters
152    let mut code = None;
153    let mut state = None;
154
155    for (key, value) in url.query_pairs() {
156        if key == "code" {
157            code = Some(value.to_string());
158        } else if key == "state" {
159            state = Some(value.to_string());
160        }
161    }
162
163    // Verify the state to prevent CSRF attacks
164    if state.as_deref() != Some(csrf_state.secret()) {
165        // Send an error response to the browser
166        let error_html = "<html><body><h1>Authentication Error</h1><p>Invalid state parameter. This could be a CSRF attack.</p></body></html>";
167        req.respond(Response::from_string(error_html))?;
168
169        return Err(anyhow!("Invalid OAuth state"));
170    }
171
172    // Check for the code and respond appropriately
173    match code {
174        Some(code_value) => {
175            // Send a success response to the browser
176            let success_html = "<h1>Authentication Successful!</h1><p>You can now close this window and return to the terminal.</p>";
177            req.respond(Response::from_string(success_html))?;
178
179            Ok(code_value)
180        }
181        None => {
182            // Send an error response for missing code
183            let error_html =
184                "<h1>Authentication Error</h1><p>No authorization code received from GitHub.</p>";
185            req.respond(Response::from_string(error_html))?;
186
187            Err(anyhow!("No authorization code received"))
188        }
189    }
190}

192/// Gets the GitHub username from the user's profile
193async fn get_github_username(token: &str) -> Result<String> {
194    let client = reqwest::Client::new();
195    let user: GitHubUser = client
196        .get("https://api.github.com/user")
197        .header("User-Agent", "faasta-cli")
198        .header("Authorization", format!("Bearer {}", token))
199        .send()
200        .await?
201        .json()
202        .await?;
203
204    Ok(user.login)
205}
