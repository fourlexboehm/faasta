64// Create a basic response with string content
65fn text_response(status: u16, text: &str) -> Result<Response<HyperOutgoingBody>> {
66    // Create a simple body with the provided text
67    // Clone the text to ensure it's owned data that will live beyond this function
68    let text_owned = text.to_string();
69    let body = Full::new(Bytes::from(text_owned))
70        .map_err(|_| ErrorCode::InternalError(None))
71        .boxed();
72
73    // Build and return the response
74    Ok(Response::builder()
75        .status(status)
76        .header("Content-Type", "text/plain")
77        .body(HyperOutgoingBody::new(body))?)
78}

82    // Create a simple body with a redirect message

84    let body = Full::new(Bytes::from(text_owned))
85        .map_err(|_| ErrorCode::InternalError(None))

88    // Build and return the redirect response
89    Ok(Response::builder()
90        .status(302)
91        .header("Location", "https://website.faasta.xyz")
92        .header("Content-Type", "text/plain")
93        .body(HyperOutgoingBody::new(body))?)

182// Implement request handling methods on RequestContext
183impl RequestContext {
184    async fn handle_request(&self, req: Request<Incoming>) -> Result<Response<HyperOutgoingBody>> {
185        // Extract function name from subdomain or path
186        let host_header = req
187            .headers()
188            .get(HOST)
189            .and_then(|h| h.to_str().ok())
190            .map(|s| s.to_string());
191        let path = req.uri().path().to_string();
192
193        debug!("Handling request with path: {}", path);
194
195        // Check if it's the root domain or local development host
196        if host_header
197            .as_deref()
198            .map(|h| {
199                h == self.base_domain || h.starts_with("localhost") || h.starts_with("127.0.0.1")
200            })
201            .unwrap_or(false)
202        {
203            debug!("Processing request on root domain: {}", self.base_domain);
204            // Root domain with no subdomain - try to route based on path
205            let path_str = path.as_str();
206            let path_parts: Vec<&str> = path_str.split('/').collect();
207
208            if path_parts.len() >= 2 && !path_parts[1].is_empty() {
209                let function_name = path_parts[1].to_string();
210                debug!(
211                    "Processing path-based request for function: {}",
212                    function_name
213                );
214
215                // Use direct function name approach
216                let wasm_filename = format!("{}.wasm", function_name);
217                debug!("Looking for WASM file: {}", wasm_filename);
218
219                // Create a timer for this function call - will be moved to execute_function
220                let function_path = self.functions_dir.join(&wasm_filename);
221
222                // Debug logging to track function path
223                if function_path.exists() {
224                    debug!("Found function at path: {:?}", function_path);
225                    // Create a new path to remove the /{function_name} prefix
226                    let new_path = if path_parts.len() > 2 {
227                        // Keep the rest of the path
228                        format!("/{}", path_parts[2..].join("/"))
229                    } else {
230                        // Just the root
231                        "/".to_string()
232                    };
233
234                    debug!("Rewriting path to: {}", new_path);
235
236                    // Build a new request with the modified path
237                    let mut builder = Request::builder()
238                        .method(req.method().clone())
239                        .uri(new_path)
240                        .version(req.version());
241
242                    // Copy all headers
243                    for (name, value) in req.headers() {
244                        builder = builder.header(name, value);
245                    }
246
247                    let (_, body) = req.into_parts();
248                    let new_req = builder.body(body)?;
249
250                    return self
251                        .execute_function(new_req, &function_name, &function_path)
252                        .await;
253                } else {
254                    debug!("Function not found at path: {:?}", function_path);
255                    // If we're looking for a specific function but it doesn't exist, return a 404
256                    return text_response(404, &format!("Function '{}' not found", function_name));
257                }
258            }
259
260            // No function found in path, redirect to website
261            debug!("No function specified in path, redirecting to website");
262            return redirect_to_website();
263        }
264
265        if let Some(host) = &host_header {
266            let expected_suffix = format!(".{}", self.base_domain);
267            debug!("Checking host: {} for subdomain routing", host);
268
269            if !host
270                .to_lowercase()
271                .ends_with(&expected_suffix.to_lowercase())
272                && !host.starts_with("localhost")
273                && !host.starts_with("127.0.0.1")
274            {
275                debug!(
276                    "Host doesn't end with expected suffix: {} and is not a local development host",
277                    expected_suffix
278                );
279                return redirect_to_website();
280            }
281
282            // Use string view instead of cloning
283            let subdomain = host.trim_end_matches(&expected_suffix);
284            if subdomain.is_empty() || subdomain == *host {
285                debug!("Empty subdomain or hostname equals subdomain, redirecting");
286                return redirect_to_website();
287            }
288
289            debug!("Processing subdomain request for function: {}", subdomain);
290
291            // Use direct function name approach - only format once
292            let wasm_filename = format!("{}.wasm", subdomain);
293            debug!("Looking for WASM file: {}", wasm_filename);
294
295            // Create a timer for this function call - will be moved to execute_function
296            let function_path = self.functions_dir.join(&wasm_filename);
297            if !function_path.exists() {
298                debug!("Function not found at path: {:?}", function_path);
299                return text_response(404, &format!("Function '{}' not found", subdomain));
300            }
301
302            // Execute the function
303            debug!("Executing function from subdomain route");
304            return self.execute_function(req, subdomain, &function_path).await;
305        } else {
306            // No host header, redirect to website
307            debug!("No host header found, redirecting to website");
308            redirect_to_website()
309        }
310    }
311}

313// Implement function execution methods on RequestContext
314impl RequestContext {
315    async fn execute_function(
316        &self,
317        req: Request<Incoming>,
318        function_name: &str,
319        function_path: &PathBuf,
320    ) -> Result<Response<HyperOutgoingBody>> {
321        let _timer = Timer::new(function_name.to_string());
322
323        debug!(
324            "Executing function: {} [path: {:?}]",
325            function_name, function_path
326        );
327
328        // Initialize a store template function if not already done
329        let store_template = STORE_TEMPLATE_CTX.get_or_init(|| {
330            // This template function will be used to create a similarly configured store each time
331            Box::new(move || {
332                FaastaClientState {
333                    table: ResourceTable::new(),
334                    wasi: WasiCtxBuilder::new().inherit_stdio().build(),
335                    http: WasiHttpCtx::new(),
336                }
337            })
338        });
339
340        // Use the template to create a store with similar configuration
341        let mut client_state = store_template();
342
343
344        // Update environment for this specific function
345        client_state.wasi = WasiCtxBuilder::new()
346            // .inherit_stdio()
347            .env("FUNCTION_NAME", function_name)
348            .build();
349
350        // Get or load the ProxyPre
351        let pre = self
352            .get_or_load_proxy_pre(function_name, function_path)
353            .await?;
354
355        // Create store with client state
356        let mut store = Store::new(pre.engine(), client_state);
357
358        // Setup the response channel
359        let (sender, receiver) = tokio::sync::oneshot::channel();
360
361        // Create the WASI HTTP request
362        let wasi_req = store.data_mut().new_incoming_request(Scheme::Http, req)?;
363        let wasi_resp_out = store.data_mut().new_response_outparam(sender)?;
364
365        let proxy = pre.instantiate_async(&mut store).await?;
366
367        // Spawn a task to handle the function execution
368        let task = tokio::task::spawn(async move {
369            proxy
370                .wasi_http_incoming_handler()
371                .call_handle(store, wasi_req, wasi_resp_out)
372                .await?;
373            Ok::<_, anyhow::Error>(())
374        });
375
376        // Wait for response with a 10-minute timeout
377        match tokio::time::timeout(std::time::Duration::from_secs(600), receiver).await {
378            Ok(receiver_result) => match receiver_result {
379                Ok(Ok(resp)) => Ok(resp),
380                Ok(Err(err_code)) => {
381                    error!("Function returned error: {:?}", err_code);
382                    Err(anyhow!("Function error: {:?}", err_code))
383                }
384                Err(_) => match task.await {
385                    Ok(Ok(())) => bail!("Function did not set response"),
386                    Ok(Err(e)) => Err(e),
387                    Err(e) => Err(e.into()),
388                },
389            },
390            Err(_) => {
391                error!("Function execution timed out after 10 minutes");
392                Err(anyhow!("Function execution timed out after 10 minutes"))
393            }
394        }
395    }
396
397    async fn get_or_load_proxy_pre(
398        &self,
399        function_name: &str,
400        function_path: &PathBuf,
401    ) -> Result<ProxyPre<FaastaClientState>> {
402        // First check if we have this pre-cached without cloning the key
403        if let Some(cached) = self.pre_cache.get(function_name) {
404            return Ok(cached.value().clone());
405        }
406
407        // Get the component, either from cache or newly loaded
408        let component = if let Some(cached) = self.component_cache.get(function_name) {
409            cached.value().clone()
410        } else {
411            // Load the component
412            let component = Component::from_file(&self.engine, function_path)?;
413
414            // Cache the component
415            self.component_cache
416                .insert(function_name.to_string(), component.clone());
417
418            component
419        };
420
421        // Get the shared linker or create it once
422        let linker = SHARED_LINKER.get_or_init(|| {
423            let mut linker = Linker::new(&self.engine);
424
425            // Set up WASI and WASI-HTTP definitions - only needs to be done once
426            wasmtime_wasi::add_to_linker_with_options_async(&mut linker, &LinkOptions::default())
427                .expect("Failed to add WASI to linker");
428            wasmtime_wasi_http::add_only_http_to_linker_async(&mut linker)
429                .expect("Failed to add WASI-HTTP to linker");
430
431            linker
432        });
433
434        let pre = ProxyPre::new(linker.instantiate_pre(&component)?)?;
435
436        // Cache it for future use - only clone the string once when inserting
437        self.pre_cache
438            .insert(function_name.to_string(), pre.clone());
439
440        Ok(pre)
441    }
442}

495async fn load_tls_config(args: &Args) -> Result<Arc<ServerConfig>> {
496    // Load TLS certificate
497    let cert_file = File::open(&args.tls_cert_path)
498        .with_context(|| format!("Failed to open TLS cert file: {:?}", args.tls_cert_path))?;
499    let mut cert_reader = BufReader::new(cert_file);
500    let certs = certs(&mut cert_reader)
501        .collect::<Result<Vec<_>, _>>()
502        .context("Failed to parse TLS certificates")?;
503
504    // Load TLS private key
505    let key_file = File::open(&args.tls_key_path)
506        .with_context(|| format!("Failed to open TLS key file: {:?}", args.tls_key_path))?;
507    let mut key_reader = BufReader::new(key_file);
508    let key = private_key(&mut key_reader)?.context("No private key found in TLS key file")?;
509
510    // Build TLS config
511    let config = ServerConfig::builder()
512        .with_no_client_auth()
513        .with_single_cert(certs, key)
514        .context("Failed to build TLS server config")?;
515
516    Ok(Arc::new(config))
517}

519// Function to handle connections for tarpc
520async fn run_rpc_server(mut quic_server: s2n_quic::Server, server: Arc<FaastaServer>) {
521    while let Some(mut connection) = quic_server.accept().await {
522        // Clone the server Arc once for each connection
523        let server_clone = server.clone();
524
525        tokio::spawn(async move {
526            debug!("Accepted new connection");
527
528            while let Ok(Some(stream)) = connection.accept_bidirectional_stream().await {
529                // Create a new service for each stream using the connection's server_clone
530                let functions_dir = server_clone.functions_dir.clone();
531                let github_auth = server_clone.github_auth.clone();
532                let metadata_db = server_clone.metadata_db.clone();
533
534                tokio::spawn(async move {
535                    debug!("Accepted new stream");
536                    let framed = LengthDelimitedCodec::builder().new_framed(stream);
537                    let transport = transport::new(framed, Bincode::default());
538
539                    let service = rpc_service::create_service_with_github_auth(
540                        functions_dir,
541                        github_auth,
542                        metadata_db,
543                    )
544                    .expect("Failed to create function service");
545
546                    // Process this connection
547                    // Use default configuration but with a longer context deadline
548                    let server_channel = BaseChannel::with_defaults(transport);
549
550                    // Use a reference to the service to call serve()
551                    server_channel
552                        .execute(service.serve())
553                        .for_each(|fut| {
554                            tokio::spawn(fut);
555                            async {}
556                        })
557                        .await;
558                });
559            }
560        });
561    }
562}

564// HTTP to HTTPS redirection using Axum framework
565async fn run_http_redirect_server(http_listener: TcpListener) {
566    info!("HTTP redirect server listening for connections");
567
568    // Create a function to convert HTTP URLs to HTTPS
569    let make_https = |host: &str, uri: Uri, https_port: u16| -> Result<Uri, BoxError> {
570        let mut parts = uri.into_parts();
571
572        parts.scheme = Some(axum::http::uri::Scheme::HTTPS);
573
574        if parts.path_and_query.is_none() {
575            parts.path_and_query = Some("/".parse().unwrap());
576        }
577
578        let authority: Authority = host.parse()?;
579        let bare_host = match authority.port() {
580            Some(port_struct) => authority
581                .as_str()
582                .strip_suffix(port_struct.as_str())
583                .unwrap()
584                .strip_suffix(':')
585                .unwrap(), // if authority.port() is Some(port) then we can be sure authority ends with :{port}
586            None => authority.as_str(),
587        };
588
589        parts.authority = Some(format!("{bare_host}:{https_port}").parse()?);
590
591        Ok(Uri::from_parts(parts)?)
592    };
593
594    // Get the local port this listener is bound to
595    let listener_addr = http_listener.local_addr().unwrap();
596
597    // Determine HTTPS port (default to 443)
598    let https_port = 443;
599
600    // Create the redirect handler
601    let redirect = move |Host(host): Host, uri: Uri| async move {
602        match make_https(&host, uri, https_port) {
603            Ok(uri) => Ok(Redirect::permanent(&uri.to_string())),
604            Err(error) => {
605                tracing::warn!(%error, "failed to convert URI to HTTPS");
606                Err(axum::http::StatusCode::BAD_REQUEST)
607            }
608        }
609    };
610
611    // Create Axum router with the redirect handler
612    let app = Router::new().fallback(redirect);
613
614    // Start the Axum HTTP server
615    info!(
616        "HTTP redirect service listening on http://{}",
617        listener_addr
618    );
619
620    // Serve with the existing TcpListener
621    axum::serve(http_listener, app).await.unwrap();
622}

665
666        // Find all .wasm files in the functions directory
667        let function_files = std::fs::read_dir(functions_dir)?
668            .filter_map(|entry| {
669                let entry = entry.ok()?;
670                let path = entry.path();
671                if path.extension().and_then(|ext| ext.to_str()) == Some("wasm") {
672                    Some(path)
673                } else {
674                    None
675                }
676            })
677            .collect::<Vec<_>>();
678
679        // Log how many functions we're going to precompile
680        info!("Found {} functions to precompile", function_files.len());
681
682        // Precompile each function
683        for path in function_files {
684            let filename = path.file_name().unwrap().to_string_lossy();
685            info!("Precompiling function: {}", filename);
686            let _ = Component::from_file(engine, &path)?;
687        }
688
689        info!("Precompilation complete");
690        Ok(())
691    }
692
693    // Create a clone for use in the QUIC server task
694    let args_clone = args.clone();
695
696    info!(
697        "Starting server-wasi with base domain: {}",
698        args.base_domain
699    );
700
701    // Ensure metrics database directory exists
702    let metrics_db_path =
703        std::env::var("METRICS_DB_PATH").unwrap_or_else(|_| "./data/metrics".to_string());
704    std::fs::create_dir_all(
705        std::path::Path::new(&metrics_db_path)
706            .parent()
707            .unwrap_or_else(|| std::path::Path::new(".")),
708    )?;
709
710    // Open/create component cache database
711    let pre_cache = sled::open(&args.db_path)?;
712
713    // Configure Wasmtime engine
714    let cache_dir =
715        std::env::var("WASMTIME_CACHE_DIR").unwrap_or_else(|_| "./data/wasmtime-cache".to_string());
716    std::fs::create_dir_all(&cache_dir)?;
717    info!("Using wasmtime cache directory: {}", cache_dir);
718
719    let mut config = Config::default();
720    config.async_support(true);
721    config.wasm_component_model(true);
722
723    // Enable module caching to speed up startup time
724    config.cache_config_load_default()?;
725
726    // Set compilation settings
727    config.cranelift_opt_level(wasmtime::OptLevel::Speed);
728
729    // Enable parallel compilation if available
730    config.parallel_compilation(true);
731
732    // Precompile modules ahead of time
733    config.strategy(wasmtime::Strategy::Cranelift);
734
735    // Create the engine
736    let engine = Engine::new(&config)?;
737
738    // // Precompile functions
739    // if let Err(e) = precompile_functions(&engine, &args.functions_path).await {
740    //     error!("Error precompiling functions: {}", e);
741    // }
742
743    // Create server
744    let server_instance = Arc::new(
745        FaastaServer::new(
746            engine,
747            pre_cache,
748            args.base_domain.clone(),
749            args.functions_path.clone(),
750        )
751        .await?,
752    );
753
754    // Store server in global OnceCell for cache management
755    let _ = SERVER.set(server_instance.clone());
756
757    // Spawn a background task to flush metrics to DB every hour
758    metrics::spawn_periodic_flush(60 * 30);
759
760    // Load TLS configuration
761    let tls_config = load_tls_config(&args).await?;
762
763    let tls_acceptor = TlsAcceptor::from(tls_config.clone());
764
765    // Start listening for HTTP connections (for redirects)
766    let http_listener = TcpListener::bind(&args.http_listen_addr)
767        .await
768        .with_context(|| format!("Failed to bind to {}", args.http_listen_addr))?;
769    info!(
770        "HTTP redirect service listening on http://{}",
771        args.http_listen_addr
772    );
773
774    // Start HTTP redirect server as a separate tokio task
775    tokio::spawn(run_http_redirect_server(http_listener));
776
777    // Start listening for HTTPS connections
778    let listener = TcpListener::bind(&args.listen_addr)
779        .await
780        .with_context(|| format!("Failed to bind to {}", args.listen_addr))?;
781    info!("Listening on https://{}", args.listen_addr);
782
783    // Start tarpc service for function management
784    let server_clone = server_instance.clone();
785    tokio::spawn(async move {
786        let addr = "0.0.0.0:4433".parse::<std::net::SocketAddr>().unwrap();
787
788        // Configure server with the TLS certs
789        let quic_server = s2n_quic::Server::builder()
790            .with_tls((
791                Path::new(&args_clone.tls_cert_path),
792                Path::new(&args_clone.tls_key_path),
793            ))
794            .map_err(|e| anyhow::anyhow!("Failed to set up TLS: {:?}", e))
795            .expect("Failed to set up TLS")
796            .with_io(addr)
797            .map_err(|e| anyhow::anyhow!("Failed to set up IO: {:?}", e))
798            .expect("Failed to set up IO")
799            .start()
800            .map_err(|e| anyhow::anyhow!("Failed to start server: {:?}", e))
801            .expect("Failed to start server");
802
803        info!("RPC service listening on {}", addr);
804
805        // Process connections
806        run_rpc_server(quic_server, server_clone).await;
807    });
808    // Main server loop
809    loop {
810        // Accept incoming connection
811        let (stream, peer_addr) = match listener.accept().await {
812            Ok(conn) => conn,
813            Err(e) => {
814                error!("Failed to accept connection: {}", e);
815                continue;
816            }
817        };
818        info!("Accepted connection from {}", peer_addr);
819
820        // Clone server and acceptor for this connection
821        let server = server_instance.clone();
822        let tls_acceptor = tls_acceptor.clone();
823
824        // Handle connection in a new task
825        tokio::spawn(async move {
826            // Perform TLS handshake
827            match tls_acceptor.accept(stream).await {
828                Ok(tls_stream) => {
829                    info!("TLS handshake successful with {}", peer_addr);
830
831                    // Create a lightweight request context once for this connection
832                    let context = server.get_request_context();
833
834                    // Create a service function for handling HTTP requests
835                    // Clone the context for each request to avoid lifetime issues
836                    let context_clone = context.clone();
837                    let service = service_fn(move |req: Request<Incoming>| {
838                        // Use the cloned context for each request
839                        let context = context_clone.clone();
840
841                        async move {
842                            match context.handle_request(req).await {
843                                Ok(response) => Ok::<_, anyhow::Error>(response),
844                                Err(e) => {
845                                    error!("Error handling request: {}", e);
846                                    // Return a generic 500 error response
847                                    match text_response(500, "Internal Server Error") {
848                                        Ok(resp) => Ok(resp),
849                                        Err(err) => {
850                                            error!("Failed to create error response: {}", err);
851                                            // Fall back to a minimal hard-coded response if everything else fails
852                                            let error_text = "Internal Server Error".to_string();
853                                            let body = Full::new(Bytes::from(error_text))
854                                                .map_err(|_| ErrorCode::InternalError(None))
855                                                .boxed();
856                                            Ok(Response::builder()
857                                                .status(500)
858                                                .header("Content-Type", "text/plain")
859                                                .body(HyperOutgoingBody::new(body))
860                                                .unwrap())
861                                        }
862                                    }
863                                }
864                            }
865                        }
866                    });
867
868                    // Serve the HTTP connection directly with hyper
869                    if let Err(err) = http1::Builder::new()
870                        .serve_connection(TokioIo::new(tls_stream), service)
871                        .await
872                    {
873                        // Only log errors that aren't from client disconnects
874                        if !err.is_closed() && !err.is_canceled() {
875                            error!("Error serving connection from {}: {}", peer_addr, err);
876                        }
877                    }
878                }
879                Err(e) => {
880                    error!("TLS handshake failed with {}: {}", peer_addr, e);
881                }
882            }
883        });
884    }
885}
