13pub async fn run_rpc_server(
14    tls_cert_path: PathBuf,
15    tls_key_path: PathBuf,
16    listen_addr: String,
17) -> Result<()> {
18    let cert_chain = load_certificates(&tls_cert_path)?;
19    let private_key = load_private_key(&tls_key_path)?;
20
21    let quic_config = QuicServerBuilder::new_with_single_cert(cert_chain, private_key)
22        .context("failed to build QUIC config")?
23        .with_alpn_protocols(&["h3"])
24        .build();
25
26    let service = rpc_service::create_service().context("failed to create RPC service")?;
27    info!("RPC service listening on {listen_addr}");
28
29    ServerBuilder::new(quic_config, listen_addr)
30        .serve(RpcRequestServiceWrapper(service))
31        .await
32        .context("bitRPC server error")
33}

35fn load_certificates(path: &PathBuf) -> Result<Vec<CertificateDer<'static>>> {
36    let file = File::open(path).with_context(|| format!("failed to open cert file: {path:?}"))?;
37    let mut reader = BufReader::new(file);
38    let certs = rustls_pemfile::certs(&mut reader)
39        .collect::<std::result::Result<Vec<_>, _>>()
40        .context("failed to read certificate chain")?;
41
42    if certs.is_empty() {
43        anyhow::bail!("no certificates found in {:?}", path);
44    }
45
46    Ok(certs)
47}
48
49fn load_private_key(path: &PathBuf) -> Result<PrivateKeyDer<'static>> {
50    let file = File::open(path).with_context(|| format!("failed to open key file: {path:?}"))?;
51    let mut reader = BufReader::new(file);
52    rustls_pemfile::private_key(&mut reader)
53        .context("failed to parse private key")?
54        .ok_or_else(|| anyhow::anyhow!("no private key found in {:?}", path))
55}
